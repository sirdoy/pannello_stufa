**Create `app/components/devices/raspi/RaspiCard.tsx`:**

Orchestrator component following NetworkCard pattern exactly.

```typescript
'use client';

import Skeleton from '../../ui/Skeleton';
import { SmartHomeCard } from '../../ui';
import { HealthIndicator, Banner } from '../../ui';
import { useRaspiData } from './hooks/useRaspiData';
import RaspiStats from './components/RaspiStats';

export default function RaspiCard() {
  const { data, loading, error, stale, health } = useRaspiData();

  // Loading state — show skeleton
  if (loading) {
    return <Skeleton.RaspiCard />;
  }

  // Error state — no cached data available
  if (error && !data) {
    return (
      <SmartHomeCard icon="🖥️" title="Raspberry Pi" colorTheme="sage">
        <SmartHomeCard.Controls>
          <Banner variant="warning" title="Non raggiungibile" compact={false}>
            <p className="text-sm text-slate-300 [html:not(.dark)_&]:text-slate-600">
              {error}
            </p>
          </Banner>
        </SmartHomeCard.Controls>
      </SmartHomeCard>
    );
  }

  // Main card with data
  return (
    <SmartHomeCard
      icon="🖥️"
      title="Raspberry Pi"
      colorTheme="sage"
      headerActions={
        <HealthIndicator
          status={health}
          size="sm"
          showIcon={true}
          label=""
        />
      }
    >
      {/* Stale banner — shows when data exists but latest fetch failed */}
      {stale && (
        <SmartHomeCard.Controls>
          <Banner variant="warning" title="Dati non aggiornati" compact={true}>
            <p className="text-xs text-slate-400">Ultimo aggiornamento non riuscito</p>
          </Banner>
        </SmartHomeCard.Controls>
      )}

      {/* Metrics grid */}
      {data && (
        <SmartHomeCard.Controls>
          <RaspiStats data={data} />
        </SmartHomeCard.Controls>
      )}
    </SmartHomeCard>
  );
}
```

Key points:
- `colorTheme="sage"` — same as NetworkCard, consistent infrastructure theme
- Loading returns `<Skeleton.RaspiCard />` directly (not wrapped in SmartHomeCard)
- Error state only shows when `error && !data` (no cached data)
- Stale banner shows when data exists but latest poll failed
- `HealthIndicator` in `headerActions` — same pattern as NetworkCard
- `RaspiStats` is purely presentational — receives `data` prop
- Guard `data.cpuTemperature !== null` in RaspiStats prevents null rendering

import { render, screen } from '@testing-library/react';
import RaspiCard from '../RaspiCard';

import type { UseRaspiDataReturn } from '../hooks/useRaspiData';

jest.mock('../hooks/useRaspiData');
const mockUseRaspiData = jest.mocked(useRaspiData);

// Mock SmartHomeCard and HealthIndicator as pass-through
jest.mock('../../../ui', () => ({
  SmartHomeCard: Object.assign(
    ({ children, title, headerActions }: any) => (

        {headerActions}
        {children}
      </div>
    ),
    { Controls: ({ children }: any) => <div>{children}</div> }
  ),
  HealthIndicator: ({ status }: any) => <div data-testid="health-indicator" data-status={status} />,
  Banner: ({ title, children }: any) => <div data-testid="banner" data-title={title}>{children}</div>,
}));

jest.mock('../../../ui/Skeleton', () => {
  const Skeleton = () => null;
  Skeleton.RaspiCard = () => <div data-testid="skeleton-raspi" />;
  Skeleton.Card = ({ children }: any) => <div>{children}</div>;
  return { __esModule: true, default: Skeleton };
});

const baseData: UseRaspiDataReturn = {
  data: { cpuPercent: 25.3, memoryPercent: 50.0, diskPercent: 25.0, cpuTemperature: 45.2 },
  loading: false,
  error: null,
  stale: false,
  health: 'ok',
};

describe('RaspiCard', () => {
  beforeEach(() => {
    mockUseRaspiData.mockReturnValue(baseData);

  it('renders 4 metric values when data available', () => {

    expect(screen.getByTestId('skeleton-raspi')).toBeInTheDocument();

    mockUseRaspiData.mockReturnValue({

    expect(screen.getByTestId('health-indicator')).toBeInTheDocument();

    mockUseRaspiData.mockReturnValue({
      ...baseData,
      data: { ...baseData.data!, cpuTemperature: null },

    render(<RaspiCard />);
    expect(screen.getByText('—')).toBeInTheDocument();
