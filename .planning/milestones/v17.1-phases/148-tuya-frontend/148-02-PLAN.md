      ```
      const activeCount = plugs.filter(p => p.switch_on === true).length;
      const inactiveCount = plugs.length - activeCount;
      const totalPowerW = plugs.reduce((sum, p) => sum + (p.power_w ?? 0), 0);
      const highestConsumer = plugs.reduce<TuyaPlug | null>((max, p) =>

      const gaugePercent = Math.min((totalPowerW / 3500) * 100, 100);
      ```

      - Power gauge bar: outer div with bg-slate-700/30 rounded-full h-2, inner div with bg-gradient-to-r from-warning-400 to-warning-600 and width={gaugePercent}%
      - Bottom: if highestConsumer and highestConsumer.power_w > 0, show "Consumo max: {highestConsumer.custom_name ?? highestConsumer.device_id} ({highestConsumer.power_w?.toFixed(0)} W)" in text-xs text-slate-400
    - Use Tailwind classes consistent with Ember Noir dark-first design
