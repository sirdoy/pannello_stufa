    ```typescript
    'use client';

    import { useState, useCallback } from 'react';

    export type LoginError =
      | 'RATE_LIMITED'
      | 'INVALID_CREDENTIALS'
      | 'SESSION_EXPIRED'
      | 'NETWORK_ERROR'
      | 'SERVER_ERROR';

    export interface LoginPayload {
      username: string;
      password: string;
    }

    export interface UseLoginReturn {
      authenticated: boolean;
      loading: boolean;
      error: LoginError | null;
      rateLimitedUntil: number;                       // Date.now() sentinel; 0 = not rate-limited
      login: (payload?: LoginPayload) => Promise<boolean>;
      logout: () => Promise<void>;
    }

    const RATE_LIMIT_WINDOW_MS = 30_000;

    export function useLogin(): UseLoginReturn {
      const [authenticated, setAuthenticated] = useState(false);
      const [loading, setLoading] = useState(false);
      const [error, setError] = useState<LoginError | null>(null);
      const [rateLimitedUntil, setRateLimitedUntil] = useState(0);

      const login = useCallback(async (payload?: LoginPayload): Promise<boolean> => {
        // Client-side lockout (D-18)
        if (rateLimitedUntil > Date.now()) {
          setError('RATE_LIMITED');
          return false;
        }

        setLoading(true);
        setError(null);

        try {
          const res = await fetch('/api/auth/login', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(payload ?? {}),
          });

          if (res.ok) {
            setAuthenticated(true);
            return true;
          }

          if (res.status === 429) {
            setError('RATE_LIMITED');
            setRateLimitedUntil(Date.now() + RATE_LIMIT_WINDOW_MS);
            return false;
          }
          if (res.status === 401) {
            setError('INVALID_CREDENTIALS');
            return false;
          }
          setError('SERVER_ERROR');
          return false;
        } catch {
          setError('NETWORK_ERROR');
          return false;
        } finally {
          setLoading(false);
        }
      }, [rateLimitedUntil]);

      const logout = useCallback(async (): Promise<void> => {
        setLoading(true);
        try {
          await fetch('/api/auth/logout', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: '{}',
          });
          setAuthenticated(false);
          setError(null);
        } catch {
          setError('NETWORK_ERROR');
        } finally {
          setLoading(false);
        }
      }, []);

      return { authenticated, loading, error, rateLimitedUntil, login, logout };
    }
    ```

    ```typescript
    import { useState, useCallback, useEffect } from 'react';
    import type { APIKeyInfo, APIKeyResponse, APIKeyListResponse } from '@/types/authProxy';

    export type ApiKeysError = 'SESSION_EXPIRED' | 'Errore nel caricamento delle API key';

    export interface UseApiKeysReturn {
      keys: APIKeyInfo[];
      loading: boolean;
      error: ApiKeysError | null;
      refetch: () => Promise<void>;
      create: (name: string) => Promise<APIKeyResponse>;
      revoke: (id: number) => Promise<void>;
    }

    export function useApiKeys(): UseApiKeysReturn {
      const [keys, setKeys] = useState<APIKeyInfo[]>([]);
      const [loading, setLoading] = useState(true);
      const [error, setError] = useState<ApiKeysError | null>(null);

      const refetch = useCallback(async (): Promise<void> => {
        setLoading(true);
        setError(null);
        try {
          const res = await fetch('/api/auth/api-keys', { method: 'GET' });
          if (res.status === 401) {
            setError('SESSION_EXPIRED');
            setKeys([]);
            return;
          }
          if (!res.ok) {
            setError('Errore nel caricamento delle API key');
            return;
          }
          const data = (await res.json()) as APIKeyListResponse;
          setKeys(data.keys);
        } catch {
          setError('Errore nel caricamento delle API key');
        } finally {
          setLoading(false);
        }
      }, []);

      useEffect(() => { void refetch(); }, [refetch]);

      const create = useCallback(async (name: string): Promise<APIKeyResponse> => {
        const res = await fetch('/api/auth/api-keys', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ name }),
        });
        if (!res.ok) {
          throw new Error('Errore durante la creazione della chiave');
        }
        const wrapped = (await res.json()) as { data: APIKeyResponse };
        return wrapped.data;
      }, []);

      const revoke = useCallback(async (id: number): Promise<void> => {
        const res = await fetch(`/api/auth/api-keys/${id}`, { method: 'DELETE' });
        if (!res.ok && res.status !== 404) {
          throw new Error('Errore durante la revoca');
        }
        // 204 or 404 (already-revoked) — refetch list to reflect server state
        await refetch();
      }, [refetch]);

      return { keys, loading, error, refetch, create, revoke };
    }
    ```

    ```typescript
    const loginSchema = z.object({

    });

    const FALLBACK_REDIRECT = '/settings/api-keys';

    export default function LoginPage() {
      const router = useRouter();
      const searchParams = useSearchParams();

      const [now, setNow] = useState(Date.now());
      useEffect(() => {
        if (rateLimitedUntil <= Date.now()) return;
        const id = setInterval(() => setNow(Date.now()), 1000);
        return () => clearInterval(id);

        if (error === 'SERVER_ERROR')        toastError('Errore del server. Riprova più tardi.');
      }, [error, toastError, toastWarning]);

      const onSubmit = async (values: LoginForm) => {
        const ok = await login(values);
        if (ok) {
          // Open-redirect protection: accept `next` only if it starts with `/`
          const next = searchParams.get('next');
          const target = next && next.startsWith('/') && !next.startsWith('//') ? next : FALLBACK_REDIRECT;
          router.push(target);

                      label="Username"
                      autoComplete="username"
                      autoFocus
                      {...field}
                      error={fieldState.error?.message}

                      type="password"
                      autoComplete="current-password"
                      {...field}
                      error={fieldState.error?.message}

                variant="ember"
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser form → `useLogin.login()` | User-typed credentials enter React state then are POSTed as JSON. Zod validates min-length client-side; server validates independently. |
| `?next=` query param → `router.push()` | Untrusted URL destination. Must be validated as local-path (starts with `/`) to prevent open-redirect. |
| `useApiKeys.create()` return value → reveal modal (Wave 3) | Plaintext API key passes through hook return into page state. The hook does NOT retain the plaintext — it is only in the function call return value and the caller's state. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-170-10 | Tampering (open redirect) | `/login` `?next=` param | mitigate | Handler in `app/login/page.tsx` accepts `next` ONLY if it passes `next.startsWith('/') && !next.startsWith('//')` — rejecting both absolute URLs and protocol-relative (`//evil.com`) destinations. Implementation: `const target = next && next.startsWith('/') && !next.startsWith('//') ? next : FALLBACK_REDIRECT;`. Two tests assert this: "refuses ?next when it does not start with /" (absolute URL) and "rejects protocol-relative URL in ?next= param" (`//evil.com` → `/settings/api-keys`). |
| T-170-11 | Information Disclosure (credentials in DevTools) | `useLogin.login()` POST body | accept | Passwords are unavoidably in the request body in cleartext (HTTPS encrypts in transit). Browser DevTools will show the Network request body — this is a fundamental auth-form constraint and not a code-level fix. Mitigations: httpOnly `ha_auth` cookie means the password is never re-sent after login; no credential stored in localStorage or non-httpOnly cookies. |
| T-170-12 | DoS (credential stuffing) | `useLogin.login()` | mitigate | 30s client-side lockout after first 429 (D-18). Test "lockout: second login during rate-limit window" asserts fetch is NOT called. Combined with server-side 10 req/min cap (T-170-05), effective rate is ~2 req/min per client. |
| T-170-13 | Information Disclosure (error oracle) | Login error copy | mitigate | Single generic string "Credenziali non valide" for 401 — no distinction between "user not found" and "wrong password" (HA proxy backend already returns the same 401 for both per phase 157). Server-returned error bodies are NOT rendered — hook maps to sentinels only. |
| T-170-14 | Information Disclosure (plaintext key in hook state) | `useApiKeys.create()` return | mitigate | The hook does NOT retain the plaintext. The `create()` function awaits the POST, parses `data.api_key`, and RETURNS it — the caller (Wave 3 page) is the single owner of the plaintext in React state and is responsible for wiping it on modal close (Wave 3 enforces via D-10 + test). This hook has no memory of the plaintext after the await resolves. |
| T-170-15 | Spoofing (CSRF via form POST) | `useLogin.login()`, `useApiKeys.create()`, `useApiKeys.revoke()` | mitigate | Same-origin fetch() from this SPA; combined with sameSite=lax cookie protection from Plan 01's cookie flags + Auth0's sameSite=lax appSession cookie, cross-site form POSTs cannot replay these calls. |
