        date: ts * 1000,
        online: Math.max(...records.map(r => r.online_count)),
        offline: Math.max(...records.map(r => r.offline_count)),
        total: Math.max(...records.map(r => r.total_devices)),
      })).sort((a, b) => a.date - b.date);
