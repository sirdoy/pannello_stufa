    ```typescript
    import { cookies } from 'next/headers';
    import { withAuthAndErrorHandler, success } from '@/lib/core';
    import { login } from '@/lib/auth/authProxy';
    import { ApiError, ERROR_CODES, HTTP_STATUS } from '@/lib/core/apiErrors';

    export const dynamic = 'force-dynamic';

    export const POST = withAuthAndErrorHandler(async (request) => {
      // Tolerant body parse (RESEARCH Pitfall 4: request.json() throws on empty body in Next.js 16)
      let body: { username?: string; password?: string } = {};
      const raw = await request.text();
      if (raw.trim().length > 0) {
        try {
          body = JSON.parse(raw) as { username?: string; password?: string };
        } catch {
          throw ApiError.badRequest('Invalid JSON body');
        }
      }

      const username = body.username ?? process.env.HA_ADMIN_USER;
      const password = body.password ?? process.env.HA_ADMIN_PASSWORD;

      if (!username || !password) {
        throw new ApiError(
          ERROR_CODES.EXTERNAL_API_ERROR,
          'HA proxy auth not configured: missing HA_ADMIN_USER or HA_ADMIN_PASSWORD',
          HTTP_STATUS.INTERNAL_SERVER_ERROR
        );
      }

      // T-157-01: call login for its side effect; DISCARD access_token
      await login(username, password);

      // Set ha_auth httpOnly marker cookie (D-15, flags mirror lib/auth0.ts:46-52)
      const cookieStore = await cookies();
      cookieStore.set('ha_auth', '1', {
        httpOnly: true,
        sameSite: 'lax',
        secure: process.env.NODE_ENV === 'production',
        path: '/',
        maxAge: 60 * 60,  // 1 hour
      });

      return success({ authenticated: true });
    }, 'Auth/Login');
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → Next.js API route | Untrusted JSON body (`{username, password}`) crosses here. Body is parsed with tolerant-parse (text-then-JSON.parse) so empty and malformed bodies are handled explicitly; no fields beyond username/password are extracted. |
| Next.js route → HA proxy (server-to-server) | Credentials forwarded over HTTPS in production. JWT returned by HA proxy STAYS in the route handler (`await login(...)` result is never assigned). |
| Next.js route → browser (Set-Cookie) | Only `ha_auth=1` flag cookie crosses. httpOnly prevents JS access; sameSite=lax blocks CSRF from third-party origins. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-170-01 | Information Disclosure | `app/api/auth/login/route.ts` response body | mitigate | Handler uses `await login(...)` (not `const {access_token} = await login(...)`) — the token literally has no binding. Test 7 asserts response is exactly `{ok:true, data:{authenticated:true}}` and that the stringified response contains no `access_token` substring. **Extends T-157-01.** |
| T-170-02 | Tampering (CSRF) | `POST /api/auth/login`, `POST /api/auth/logout` | mitigate | Cookie flags `sameSite: 'lax'` block cross-site form-submitted POSTs; Auth0 `withAuthAndErrorHandler` guard requires an existing Auth0 session cookie (also sameSite=lax) for every call; state-changing operations therefore require TWO sameSite=lax cookies from the origin. |
| T-170-03 | Spoofing (session fixation) | `ha_auth` cookie | mitigate | Cookie value is a static `'1'` but scope is identity-by-existence only — the server never treats the cookie value as authoritative for user identity (HA JWT is re-fetched server-side per request per phase 157 D-03). Login route OVERWRITES any existing cookie (cookies().set), so a pre-set `ha_auth` from a previous session cannot be reused. Logout route DELETES explicitly. |
| T-170-04 | Information Disclosure | Accidental logging of body fields | mitigate | Handler never `console.log`s the parsed body. `withAuthAndErrorHandler` logs ApiError metadata only (code/status/label) via `handleError`; no request body capture. |
| T-170-05 | DoS (brute-force credential stuffing) | `POST /api/auth/login` | transfer (to HA proxy) + mitigate (client-side in Plan 02) | HA proxy enforces 10 req/min per spec and returns 429 with ApiError(RATE_LIMITED/EXTERNAL_API_ERROR). Test 8 asserts the 429 propagates cleanly to the client. A 30s client-side lockout (Plan 02) is a second layer. |
| T-170-06 | Elevation of Privilege | Unauthenticated POST to `/api/auth/login` or `/api/auth/logout` | mitigate | Both handlers wrapped in `withAuthAndErrorHandler` → an active Auth0 `appSession` is a prerequisite. T-157-06 preserved. |
| T-170-07 | Tampering (malformed body) | Body parsing | mitigate | Tolerant parse catches `JSON.parse` exceptions and converts them to `ApiError.badRequest('Invalid JSON body')` (covered by Test 9). Raw body length is whatever `request.text()` yields — Next.js enforces its own body-size limits upstream (no custom handling required here). |
| T-170-08 | Information Disclosure (cookie read from JS) | `ha_auth` cookie | mitigate | `httpOnly: true` flag is browser-enforced; `document.cookie` returns empty string for this cookie. Test 6 asserts the flag is set correctly. |
| T-170-09 | Tampering (insecure transport) | Cookie over HTTP in production | mitigate | `secure: process.env.NODE_ENV === 'production'` — matches `lib/auth0.ts:49` pattern. In dev, HTTP is acceptable; in production, browser refuses to send over HTTP. |
