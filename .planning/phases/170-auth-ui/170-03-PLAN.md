    // Prefix-based active check for global sections (Registro, Stanze, API Keys, etc.)
    // For single-segment routes (/registry/types → /registry), derive prefix from first segment.
    // For two-segment routes (/settings/api-keys), match the full route as prefix.
    const isGlobalActive = (route: string) => {
      const segments = route.split('/').filter(Boolean);
      // Two-segment routes need exact-or-deeper match to avoid lighting up on sibling pages
      if (segments.length >= 2) {

      const prefix = '/' + segments[0];
      return pathname === prefix || pathname.startsWith(prefix + '/');
    };
