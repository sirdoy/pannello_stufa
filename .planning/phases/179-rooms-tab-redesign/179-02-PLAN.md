export interface MiniButtonProps {
  Icon?: LucideIcon;
  label?: string;
  filled?: boolean;
  tone?: string;
  onClick?: () => void;
  disabled?: boolean;
  ariaLabel?: string;
}

export function MiniButton({ Icon, label, filled, tone = 'var(--accent)', onClick, disabled, ariaLabel }: MiniButtonProps): JSX.Element {
  const baseStyle: CSSProperties = {
    flex: 1,
    height: 34,
    borderRadius: 10,
    display: 'inline-flex',
    alignItems: 'center',
    justifyContent: 'center',
    gap: 6,
    fontSize: 13,
    fontWeight: 500,
    cursor: disabled ? 'not-allowed' : 'pointer',
    opacity: disabled ? 0.5 : 1,
    transition: 'all 220ms cubic-bezier(.34,1.56,.64,1)',
    background: filled ? `color-mix(in oklab, ${tone} 22%, transparent)` : 'rgba(255,255,255,0.04)',
    border: filled ? `0.5px solid color-mix(in oklab, ${tone} 35%, transparent)` : '0.5px solid rgba(255,255,255,0.06)',
    color: filled ? tone : '#fff',
    boxShadow: filled ? `0 0 12px color-mix(in oklab, ${tone} 30%, transparent)` : undefined,
  };
  return (
    <button
      type="button"
      style={baseStyle}
      onClick={disabled ? undefined : onClick}
      disabled={disabled}
      aria-label={ariaLabel ?? label ?? 'button'}
    >
      {Icon ? <Icon size={14} strokeWidth={2} /> : null}
      {label ? <span>{label}</span> : null}
    </button>
  );
}

```tsx
'use client';
import type { LucideIcon } from 'lucide-react';
import type { CSSProperties, MouseEvent } from 'react';

export interface SliderRowProps {
  label: string;
  value: number;
  unit?: string;
  min?: number;
  max?: number;
  tone?: string;
  Icon?: LucideIcon;
  disabled?: boolean;
  onChange?: (next: number) => void;
}

export function SliderRow({
  label,
  value,
  unit = '',
  min = 0,
  max = 100,
  tone = 'var(--accent)',
  Icon,
  disabled,
  onChange,
}: SliderRowProps): JSX.Element {
  const percent = Math.max(0, Math.min(100, ((value - min) / (max - min)) * 100));
  const interactive = !disabled && typeof onChange === 'function';

  function handleTrackClick(e: MouseEvent<HTMLDivElement>) {
    if (!interactive) return;
    const rect = e.currentTarget.getBoundingClientRect();
    const x = Math.max(0, Math.min(rect.width, e.clientX - rect.left));
    const fraction = x / rect.width;
    const next = Math.round(min + fraction * (max - min));
    onChange?.(next);
  }

  const containerStyle: CSSProperties = {
    background: 'rgba(255,255,255,0.04)',
    border: '0.5px solid rgba(255,255,255,0.06)',
    borderRadius: 12,
    padding: 12,
    opacity: disabled ? 0.45 : 1,
  };

  const trackStyle: CSSProperties = {
    position: 'relative',
    height: 8,
    borderRadius: 4,
    background: 'rgba(255,255,255,0.06)',
    overflow: 'hidden',
    marginTop: 8,
    cursor: interactive ? 'pointer' : 'default',
  };

  const fillStyle: CSSProperties = {
    width: `${percent}%`,
    height: '100%',
    background: `linear-gradient(90deg, color-mix(in oklab, ${tone} 50%, transparent) 0%, ${tone} 100%)`,
  };

  return (
    <div style={containerStyle} aria-disabled={disabled ? 'true' : undefined}>
      <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center' }}>
        <div style={{ display: 'flex', alignItems: 'center', gap: 6 }}>
          {Icon ? <Icon size={14} strokeWidth={2} color="var(--text-2)" /> : null}
          <span style={{ fontSize: 10, color: 'var(--text-2)', textTransform: 'uppercase', letterSpacing: 1 }}>{label}</span>
        </div>
        <span style={{ fontFamily: 'var(--font-display)', fontSize: 14, color: '#fff', fontVariantNumeric: 'tabular-nums' }}>
          {value}{unit}
        </span>
      </div>
      <div style={trackStyle} onClick={handleTrackClick} role={interactive ? 'slider' : undefined} aria-valuenow={interactive ? value : undefined} aria-valuemin={interactive ? min : undefined} aria-valuemax={interactive ? max : undefined}>
        <div style={fillStyle} />
      </div>
    </div>
  );
}
```
