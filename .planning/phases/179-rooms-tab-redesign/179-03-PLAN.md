```tsx
'use client';
import { ICON_FOR } from './lib/rooms-config';
import type { RoomDevice } from './types';

export interface DeviceChipProps {
  device: RoomDevice;
}

export function DeviceChip({ device }: DeviceChipProps): JSX.Element {
  const Icon = ICON_FOR[device.kind];
  const tone = device.tone;
  const on = device.on;

  return (
    <div
      data-testid={`stanze-chip-${device.kind}-${device.name.toLowerCase().replace(/\s+/g, '-')}`}
      style={{
        aspectRatio: '1 / 1',
        borderRadius: 10,
        position: 'relative',
        background: on ? `color-mix(in oklab, ${tone} 18%, transparent)` : 'rgba(255,255,255,0.04)',
        border: on ? `0.5px solid color-mix(in oklab, ${tone} 35%, transparent)` : '0.5px solid rgba(255,255,255,0.06)',
        display: 'flex',
        alignItems: 'center',
        justifyContent: 'center',
        color: on ? tone : 'var(--text-2)',
      }}
    >
      <Icon size={14} strokeWidth={2} />
      {on ? (
        <span
          aria-hidden="true"
          style={{
            position: 'absolute',
            top: 3,
            right: 3,
            width: 5,
            height: 5,
            borderRadius: 99,
            background: tone,
            boxShadow: `0 0 6px ${tone}`,
          }}
        />
      ) : null}
    </div>
  );
}
```

Create `app/components/EmberGlass/rooms/RoomCard.tsx`. Compose Phase 177 `<GlassCard>` + `<CardHead>` and use the just-created `<DeviceChip>`. Do NOT manually wrap in Pressable — GlassCard.onOpen does that internally.

```tsx
'use client';
import { GlassCard } from '../GlassCard';
import { CardHead } from '../CardHead';
import { DeviceChip } from './DeviceChip';
import { ICON_FOR } from './lib/rooms-config';
import type { RoomConfig, RoomDevice } from './types';

export interface RoomCardProps {
  room: RoomConfig;
  devices: RoomDevice[];
  onOpen: () => void;
}

export function RoomCard({ room, devices, onOpen }: RoomCardProps): JSX.Element {
  const Icon = ICON_FOR[room.icon];
  const activeCount = devices.filter((d) => d.on).length;
  const total = devices.length;
  const visible = devices.slice(0, 6);
  const overflowCount = Math.max(0, total - 6);

  const countBadge = (
    <span
      style={{
        fontFamily: 'var(--font-display)',
        fontSize: 13,
        color: activeCount > 0 ? room.tone : 'var(--text-2)',
        fontVariantNumeric: 'tabular-nums',
      }}
    >
      {activeCount}/{total}
    </span>
  );

  return (
    <GlassCard tone={room.tone} onOpen={onOpen} testId={`stanze-room-${room.name.toLowerCase()}`}>
      <CardHead Icon={Icon} label={room.name} tone={room.tone} right={countBadge} />
      <div style={{ marginTop: 10 }}>
        {total === 0 ? (
          <div
            style={{
              padding: '14px 0',
              textAlign: 'center',
              fontSize: 11,
              color: 'var(--text-2)',
            }}
          >
            Nessun dispositivo
          </div>
        ) : (
          <div style={{ display: 'grid', gridTemplateColumns: 'repeat(3, 1fr)', gap: 6, alignContent: 'start' }}>
            {visible.map((d, i) => (
              <DeviceChip key={`${d.kind}-${d.name}-${i}`} device={d} />
            ))}
            {overflowCount > 0 ? (
              <div
                aria-label={`${overflowCount} altri dispositivi`}
                style={{
                  aspectRatio: '1 / 1',
                  borderRadius: 10,
                  border: '0.5px dashed rgba(255,255,255,0.18)',
                  display: 'flex',
                  alignItems: 'center',
                  justifyContent: 'center',
                  color: 'var(--text-2)',
                  fontSize: 12,
                  fontFamily: 'var(--font-display)',
                  fontVariantNumeric: 'tabular-nums',
                }}
              >
                +{overflowCount}
              </div>
            ) : null}
          </div>
        )}
      </div>
    </GlassCard>
  );
}
```
