Each branch self-fetches its commands hook (per D-39 self-fetch convention). The Sonos branch needs `useSonosFullData` (for fetchData) + `useSonosCommands`. Thermo/valve branch reads `useThermostatData` for `homeId` then calls `useThermostatCommands({ homeId, refetch })`. Lights branch uses `useLightsCommands({ fetchData, ...})` with `useLightsData()`. Important: per RESEARCH Pitfall 3, thermo/valve toggle must call `setRoomMode(roomId, on ? 'manual' : 'home')`, NOT `'on' | 'off'`.

```tsx
'use client';
import { Play, Pause } from 'lucide-react';
import type { RoomDevice } from './types';
import { InlineToggle } from '../InlineToggle';
import { useSonosFullData } from '@/app/components/devices/sonos/hooks/useSonosFullData';
import { useSonosCommands } from '@/app/components/devices/sonos/hooks/useSonosCommands';
import { useLightsData } from '@/app/components/devices/lights/hooks/useLightsData';
import { useLightsCommands } from '@/app/components/devices/lights/hooks/useLightsCommands';
import { useTuyaData } from '@/app/components/devices/tuya/hooks/useTuyaData';
import { useTuyaCommands } from '@/app/components/devices/tuya/hooks/useTuyaCommands';
import { useThermostatData } from '@/app/components/devices/thermostat/hooks/useThermostatData';
import { useThermostatCommands } from '@/app/components/devices/thermostat/hooks/useThermostatCommands';

export function DevicePrimaryControl({ device }: { device: RoomDevice }): JSX.Element {
  switch (device.kind) {
    case 'sonos':
      return <SonosControl device={device} />;
    case 'camera':
      return <CameraPill />;
    case 'sensor':
      return <SensorPill />;
    case 'light':
      return <LightToggle device={device} />;
    case 'plug':
      return <PlugToggle device={device} />;
    case 'thermo':
    case 'valve':
      return <ThermoToggle device={device} />;
    default:
      return <div style={{ width: 40 }} aria-hidden="true" />;
  }
}

function SonosControl({ device }: { device: RoomDevice }) {
  const data = useSonosFullData();
  const cmds = useSonosCommands({ fetchData: data.fetchData, setError: () => {} });
  const groupId = String(device.extra.id ?? '');
  const playing = device.on;
  return (
    <button
      type="button"
      aria-label={playing ? 'Pausa' : 'Riproduci'}
      onClick={() => (playing ? cmds.handlePause(groupId) : cmds.handlePlay(groupId))}
      style={{
        width: 40, height: 40, borderRadius: 99,
        background: playing ? '#fff' : 'rgba(255,255,255,0.08)',
        color: playing ? '#1a0f08' : '#fff',
        border: 'none',
        display: 'inline-flex',
        alignItems: 'center',
        justifyContent: 'center',
        cursor: 'pointer',
      }}
    >
      {playing ? <Pause size={16} /> : <Play size={16} />}
    </button>
  );
}

function CameraPill() {
  return (
    <div style={{ display: 'inline-flex', alignItems: 'center', gap: 6, padding: '4px 8px', borderRadius: 99, background: 'rgba(255,255,255,0.06)' }}>
      <span style={{ width: 6, height: 6, borderRadius: 99, background: 'var(--accent)', boxShadow: '0 0 6px var(--accent)' }} />
      <span style={{ fontSize: 10, color: '#fff', textTransform: 'uppercase', letterSpacing: 0.6 }}>LIVE</span>
    </div>
  );
}

function SensorPill() {
  return (
    <div style={{ padding: '4px 10px', borderRadius: 99, background: 'rgba(255,255,255,0.06)' }}>
      <span style={{ fontSize: 10, color: '#fff', textTransform: 'uppercase', letterSpacing: 0.6 }}>OK</span>
    </div>
  );
}

function LightToggle({ device }: { device: RoomDevice }) {
  const data = useLightsData();
  const cmds = useLightsCommands({ fetchData: data.fetchData });
  const groupId = String(device.extra.groupId ?? '');
  return (
    <InlineToggle
      on={device.on}
      color={device.tone}
      onChange={() => { void cmds.handleRoomToggle(groupId, !device.on); }}
      ariaLabel={`Toggle ${device.name}`}
    />
  );
}

function PlugToggle({ device }: { device: RoomDevice }) {
  const data = useTuyaData();
  const cmds = useTuyaCommands({ fetchData: data.fetchData ?? (async () => {}) });
  const id = String(device.extra.id ?? '');
  return (
    <InlineToggle
      on={device.on}
      color={device.tone}
      onChange={() => { void cmds.togglePlug(id, device.on); }}
      ariaLabel={`Toggle ${device.name}`}
    />
  );
}

function ThermoToggle({ device }: { device: RoomDevice }) {
  const data = useThermostatData();
  const homeId = data.topology?.home_id ?? '';
  const cmds = useThermostatCommands({ homeId, refetch: data.refetch });
  const roomId = String(device.extra.roomId ?? '');
  // Pitfall 3: setRoomMode mode union is 'manual' | 'home', NOT 'on' | 'off'
  return (
    <InlineToggle
      on={device.on}
      color={device.tone}
      onChange={() => {
        if (!homeId || !roomId) return;
        void cmds.setRoomMode(roomId, device.on ? 'manual' : 'home');
      }}
      ariaLabel={`Toggle ${device.name}`}
    />
  );
}
```
