'use client';
import { Minus, Plus, Power } from 'lucide-react';
import { useUser } from '@auth0/nextjs-auth0/client';
import { useVersion } from '@/app/context/VersionContext';
import { useStoveData } from '@/app/components/devices/stove/hooks/useStoveData';
import { useStoveCommands } from '@/app/components/devices/stove/hooks/useStoveCommands';
import { StatChip } from '../primitives/StatChip';
import { ControlRow } from '../primitives/ControlRow';
import { MiniButton } from '../primitives/MiniButton';
import type { RoomDevice } from '../types';

export function StoveBody({ device }: { device: RoomDevice }): JSX.Element {
  const { checkVersion } = useVersion();
  const { user } = useUser();
  const data = useStoveData({ checkVersion, userId: user?.sub });
  const cmds = useStoveCommands({ /* signature verified at impl time */ });

  const powerLevel = (device.extra.powerLevel as number | undefined) ?? data.powerLevel ?? 0;
  const fanLevel = (device.extra.fanLevel as number | undefined) ?? data.fanLevel ?? 0;

  const onMeno = () => {
    const next = Math.max(1, powerLevel - 1);
    cmds.handlePowerChange({ target: { value: String(next) } } as unknown as React.ChangeEvent<HTMLInputElement>);
  };
  const onPiu = () => {
    const next = Math.min(5, powerLevel + 1);
    cmds.handlePowerChange({ target: { value: String(next) } } as unknown as React.ChangeEvent<HTMLInputElement>);
  };
  const onPower = () => {
    if (device.on) cmds.handleShutdown();
    else if (!data.needsCleaning) cmds.handleIgnite();
  };

  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: 10 }}>
      <div style={{ display: 'flex', gap: 6 }}>
        <StatChip label="Target" value={`${powerLevel}/5`} />
        <StatChip label="Fiamma" value={String(powerLevel)} />
        <StatChip label="Ventola" value={String(fanLevel)} />
      </div>
      <ControlRow>
        <MiniButton Icon={Minus} label="Meno" onClick={onMeno} />
        <MiniButton Icon={Power} label="Power" filled={device.on} tone="var(--accent)" onClick={onPower} />
        <MiniButton Icon={Plus} label="Più" onClick={onPiu} />
      </ControlRow>
    </div>
  );
}

'use client';
import { StatChip } from '../primitives/StatChip';
import type { RoomDevice } from '../types';

function formatPower(watts: number): string {
  if (watts >= 1000) return `${(watts / 1000).toFixed(1)}kW`;
  return `${Math.round(watts)}W`;
}

export function PlugBody({ device }: { device: RoomDevice }): JSX.Element {
  const power = (device.extra.power as number | undefined) ?? 0;
  const todayKwh = (device.extra.today_kwh as number | undefined) ?? 0;
  return (
    <div style={{ display: 'grid', gridTemplateColumns: 'repeat(2, 1fr)', gap: 6 }}>
      <StatChip label="Ora" value={formatPower(power)} />
      <StatChip label="Oggi" value={`${todayKwh.toFixed(1)} kWh`} />
    </div>
  );
}
