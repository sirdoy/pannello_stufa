```tsx
'use client';
import { useEffect, useState } from 'react';
import { Minus, Plus } from 'lucide-react';
import { useThermostatData } from '@/app/components/devices/thermostat/hooks/useThermostatData';
import { useThermostatCommands } from '@/app/components/devices/thermostat/hooks/useThermostatCommands';
import { useDebounce } from '@/app/hooks/useDebounce';
import { DualTempReadout } from '../primitives/DualTempReadout';
import { ControlRow } from '../primitives/ControlRow';
import { MiniButton } from '../primitives/MiniButton';
import type { RoomDevice } from '../types';

function ThermoOrValveBody({ device }: { device: RoomDevice }): JSX.Element {
  const data = useThermostatData();
  const homeId = data.topology?.home_id ?? '';
  const refetch = data.refetch;
  const cmds = useThermostatCommands({ homeId, refetch });

  const initialTarget = (device.extra.target as number | undefined) ?? 20;
  const current = (device.extra.current as number | undefined) ?? 0;
  const roomId = String(device.extra.roomId ?? '');

  const [pending, setPending] = useState<number>(initialTarget);
  const debounced = useDebounce(pending, 500);

  useEffect(() => {
    if (!homeId || !roomId) return;       // Pitfall 8: gate on populated home_id + roomId
    if (debounced === initialTarget) return; // skip first render no-op
    void cmds.setRoomSetpoint(roomId, debounced);
  }, [debounced, homeId, roomId, initialTarget, cmds]);

  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: 10 }}>
      <DualTempReadout current={current} target={pending} tone={device.tone} />
      <ControlRow>
        <MiniButton Icon={Minus} label="−0.5°" onClick={() => setPending((v) => Math.round((v - 0.5) * 10) / 10)} />
        <MiniButton Icon={Plus} label="+0.5°" onClick={() => setPending((v) => Math.round((v + 0.5) * 10) / 10)} />
        <MiniButton label="Eco" onClick={() => { if (homeId) void cmds.setHomeMode('away'); }} />
        <MiniButton label="Auto" onClick={() => { if (homeId) void cmds.setHomeMode('schedule'); }} />
      </ControlRow>
    </div>
  );
}

export function ThermoBody({ device }: { device: RoomDevice }): JSX.Element {
  return <ThermoOrValveBody device={device} />;
}

export function ValveBody({ device }: { device: RoomDevice }): JSX.Element {
  return <ThermoOrValveBody device={device} />;
}
```

'use client';
import { useEffect, useState } from 'react';
import { useLightsData } from '@/app/components/devices/lights/hooks/useLightsData';
import { useLightsCommands } from '@/app/components/devices/lights/hooks/useLightsCommands';
import { useDebounce } from '@/app/hooks/useDebounce';
import { SliderRow } from '../primitives/SliderRow';
import type { RoomDevice } from '../types';

export function LightBody({ device }: { device: RoomDevice }): JSX.Element {
  const data = useLightsData();
  const cmds = useLightsCommands({ fetchData: data.fetchData });

  const initialBrightness = (device.extra.brightness as number | undefined) ?? 0;
  const groupId = String(device.extra.groupId ?? '');

  const [pending, setPending] = useState<number>(initialBrightness);
  const debounced = useDebounce(pending, 250);

  useEffect(() => {
    if (!groupId) return;
    if (debounced === initialBrightness) return;
    if (!device.on) return;
    void cmds.handleBrightnessChange(groupId, String(debounced)); // Pitfall 5: string
  }, [debounced, groupId, initialBrightness, device.on, cmds]);

  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: 10 }}>
      <SliderRow
        label="Luminosità"
        value={pending}
        unit="%"
        tone={device.tone}
        disabled={!device.on}
        onChange={(next) => setPending(next)}
      />
      <SliderRow
        label="Temperatura"
        value={2700}
        unit="K"
        min={2200}
        max={6500}
        tone={device.tone}
        disabled={true}    // Always — no Hue color-temp endpoint (Deferred per CONTEXT)
      />
    </div>
  );
}
