ActionsSection props (REQUIRED shape — no fallback variant):

```ts
import type { KeyedAction } from '../lib/with-key'; // canonical module created by Plan 02 (lib foundation)

export interface ActionsSectionProps {
  actions: KeyedAction[];                                          // NOT ActionItem[]
  onChange: (next: KeyedAction[]) => void;                          // NOT ActionItem[]
  mintActionKey: (action: ActionItem) => KeyedAction;               // REQUIRED — provided by parent
  onValidationChange?: (actionKey: string, isValid: boolean) => void; // string `__key`, NOT row index
}
```

Behavioral invariants ActionsSection MUST satisfy (these are the contract Plan 07's pruning useEffect relies on):

1. **`addAction(typeId)` mints a fresh `__key`:**
   ```ts
   const addAction = (typeId: ActionItem['type']) => {
     onChange([...actions, mintActionKey(defaultAction(typeId))]);
     setPickerOpen(false);
   };
   ```

2. **`updateAt(idx, next)` preserves `__key` for in-place field edits, mints a NEW `__key` on type-swap:**
   ```ts
   const updateAt = (idx: number, next: ActionItem) => {
     const prev = actions[idx];
     if (!prev) return;
     const wasTypeSwap = prev.type !== next.type;
     const merged: KeyedAction = wasTypeSwap
       ? mintActionKey(next)                  // fresh key — type changed
       : { ...next, __key: prev.__key };       // preserve key — only fields changed
     onChange(actions.map((a, i) => (i === idx ? merged : a)));
   };
   ```

3. **`removeAt(idx)` preserves every other row's `__key` verbatim:**
   ```ts
   const removeAt = (idx: number) => onChange(actions.filter((_, i) => i !== idx));
   ```

4. **`moveUp(idx)` / `moveDown(idx)` preserve every row's `__key` verbatim** (only positions change):
   ```ts
   const moveUp = (idx: number) => {
     if (idx === 0) return;
     const next = [...actions];
     [next[idx - 1], next[idx]] = [next[idx]!, next[idx - 1]!];
     onChange(next);
   };
   ```

5. **React `key` in `.map`:** use `key={a.__key}` (stable across reorder), NOT `key={i}`.

6. **Validation forwarder:** `onValidationChange={(isValid) => onValidationChange?.(a.__key, isValid)}` — emits the stable per-row key, NOT the row index. ActionRow's `onValidationChange` prop signature stays `(isValid: boolean) => void`; ActionsSection bridges to the keyed signature on the way out.

7. **Never strip `__key` here.** The parent editor (Plan 07) owns `stripKeys` and runs it just before the API call. ActionsSection only ever forwards `KeyedAction[]` upward.

This contract is the only acceptable shape — the "index-based fallback" mentioned in any earlier draft has been removed. Plan 07's pruning useEffect depends on the stable-`__key`-on-remove and fresh-`__key`-on-type-swap invariants above; without them, the BLOCKER 1 recovery fails.

// ─── 4. HttpWebhookForm — JSON validation ─────────────────────────────
export function HttpWebhookForm({ action, onChange, onValidationChange }: FormProps<HttpWebhookAction>) {
  const [jsonError, setJsonError] = useState<string | null>(null);
  // Display value: stringify existing payload object for editing
  const initialPayloadStr = action.payload ? JSON.stringify(action.payload, null, 2) : '';
  const [payloadStr, setPayloadStr] = useState(initialPayloadStr);

  const handlePayloadChange = (raw: string) => {
    setPayloadStr(raw);
    if (raw.trim() === '') {
      setJsonError(null);
      onValidationChange?.(true);
      onChange({ ...action, payload: null });
      return;
    }
    try {
      const parsed = JSON.parse(raw) as Record<string, unknown>;
      setJsonError(null);
      onValidationChange?.(true);
      onChange({ ...action, payload: parsed });
    } catch {
      setJsonError('JSON non valido');
      onValidationChange?.(false);
      // Do NOT call onChange — keep the previous payload until valid
    }
  };

  return (
    <>
      <FieldLabel htmlFor="action-url">URL</FieldLabel>
      <TextInput id="action-url" type="url" mono value={action.url} onChange={(v) => onChange({ ...action, url: v })} aria-label="URL webhook" />
      <div style={{ height: 8 }} />
      <FieldLabel>Metodo</FieldLabel>
      <SegmentedControl<'GET' | 'POST'>
        options={[{ value: 'GET', label: 'GET' }, { value: 'POST', label: 'POST' }]}
        value={action.method}
        onChange={(v) => onChange({ ...action, method: v })}
        aria-label="Metodo HTTP"
      />
      <div style={{ height: 8 }} />
      <FieldLabel htmlFor="action-payload">Payload (JSON)</FieldLabel>
      <textarea
        id="action-payload"
        value={payloadStr}
        onChange={(e) => handlePayloadChange(e.target.value)}
        aria-label="Payload JSON"
        rows={4}
        style={{
          width: '100%',
          boxSizing: 'border-box',
          minHeight: 80,
          borderRadius: 9,
          background: 'rgba(255,255,255,0.05)',
          border: '0.5px solid rgba(255,255,255,0.08)',
          color: '#fff',
          padding: 10,
          fontSize: 12,
          fontFamily: 'ui-monospace, SF Mono, monospace',
          outline: 'none',
          resize: 'vertical',
        }}
      />
      {jsonError && (
        <div style={{ fontSize: 11, fontWeight: 600, color: '#ff6676', marginTop: 6 }}>
          {jsonError}
        </div>
      )}
    </>
  );
}

'use client';

import { useState } from 'react';
import { Plus } from 'lucide-react';
import type { ActionItem } from '@/types/automations';
import type { KeyedAction } from '../lib/with-key';
import { ACTION_TYPES, defaultAction } from '../lib/automations-config';
import { ActionRow } from '../ActionRow';
import { TypeTile } from '../primitives/TypeTile';

export interface ActionsSectionProps {
  actions: KeyedAction[];
  onChange: (next: KeyedAction[]) => void;
  mintActionKey: (action: ActionItem) => KeyedAction;
  onValidationChange?: (actionKey: string, isValid: boolean) => void;
}

export function ActionsSection({
  actions, onChange, mintActionKey, onValidationChange,
}: ActionsSectionProps) {
  const [pickerOpen, setPickerOpen] = useState(false);

  // Picker tile -> append a brand-new keyed row.
  const addAction = (id: ActionItem['type']) => {
    onChange([...actions, mintActionKey(defaultAction(id))]);
    setPickerOpen(false);
  };

  // Field edit OR type swap. Preserve __key for field edits; mint a fresh __key on type change
  // so Plan 07's pruning effect drops the stale validation entry. See <actions_section_contract>.
  const updateAt = (idx: number, next: ActionItem) => {
    const prev = actions[idx];
    if (!prev) return;
    const wasTypeSwap = prev.type !== next.type;
    const merged: KeyedAction = wasTypeSwap
      ? mintActionKey(next)
      : { ...next, __key: prev.__key };
    onChange(actions.map((a, i) => (i === idx ? merged : a)));
  };

  // Remove preserves every other row's __key verbatim.
  const removeAt = (idx: number) => onChange(actions.filter((_, i) => i !== idx));

  // Reorder preserves every row's __key — only positions move.
  const moveUp = (idx: number) => {
    if (idx === 0) return;
    const next_arr = [...actions];
    [next_arr[idx - 1], next_arr[idx]] = [next_arr[idx]!, next_arr[idx - 1]!];
    onChange(next_arr);
  };
  const moveDown = (idx: number) => {
    if (idx === actions.length - 1) return;
    const next_arr = [...actions];
    [next_arr[idx], next_arr[idx + 1]] = [next_arr[idx + 1]!, next_arr[idx]!];
    onChange(next_arr);
  };

  return (
    <div>
      <div
        style={{
          fontSize: 12,
          color: 'var(--text-2)',
          marginBottom: 14,
          lineHeight: 1.5,
        }}
      >
        Quando il trigger scatta e le condizioni sono verificate, queste azioni vengono eseguite in
        sequenza.
      </div>

        <div
          style={{
            padding: 24,
            borderRadius: 14,
            background: 'rgba(255,255,255,0.03)',
            border: '0.5px dashed rgba(255,255,255,0.12)',
            color: 'var(--text-2)',
            fontSize: 13,
            textAlign: 'center',
            marginBottom: 12,
          }}
        >
          Nessuna azione. Aggiungine almeno una.
        </div>
      )}

      {actions.length > 0 && (
        <div style={{ display: 'flex', flexDirection: 'column', gap: 10, marginBottom: 12 }}>
          {actions.map((a, i) => (
            <ActionRow
              key={a.__key}
              action={a}
              index={i}
              total={actions.length}
              onChange={(next) => updateAt(i, next)}
              onRemove={() => removeAt(i)}
              onMoveUp={() => moveUp(i)}
              onMoveDown={() => moveDown(i)}
              onValidationChange={(isValid) => onValidationChange?.(a.__key, isValid)}
            />
          ))}
        </div>
      )}

      {pickerOpen ? (
        <div
          style={{
            background: 'rgba(255,255,255,0.04)',
            border: '0.5px solid rgba(255,255,255,0.08)',
            padding: 10,
            borderRadius: 14,
          }}
        >
          <div
            style={{
              fontSize: 11,
              color: 'var(--text-2)',
              marginBottom: 8,
              fontWeight: 600,
              textTransform: 'uppercase',
              letterSpacing: 0.8,

          >
            Scegli tipo azione
          </div>

              const Icon = a.Icon;

                  key={a.id}
                  icon={<Icon size={13} />}
                  label={a.label}
                  tone={a.tone}
                  selected={false}
                  onClick={() => addAction(a.id)}
                  aria-label={a.label}
                />
              );
            })}
          </div>
          <button
            type="button"
            onClick={() => setPickerOpen(false)}
            style={{
              marginTop: 10,
              width: '100%',
              padding: '8px 0',
              borderRadius: 9,
              border: '0.5px solid rgba(255,255,255,0.1)',
              background: 'rgba(255,255,255,0.03)',
              color: 'var(--text-2)',
