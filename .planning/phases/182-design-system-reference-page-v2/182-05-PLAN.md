    ```tsx
      for (const name of TOKEN_NAMES) init[name] = '';
      return init;

    useEffect(() => {
      const cs = getComputedStyle(document.documentElement);
      const next = {} as Record<TokenName, string>;

      }
      setTokens(next);
    }, []);
    ```
